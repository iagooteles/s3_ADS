#1. Fácil
#Questão:
#Liste todos os usuários que possuem pelo menos uma conta com saldo maior que 1000.


SELECT *
FROM usuario u
WHERE u.codigo IN (
        SELECT DISTINCT codigo FROM conta c WHERE c.saldo > 1000
);
